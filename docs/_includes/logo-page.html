          object Union {
        case class A()
        case class B()
        case class C()
        
        def union(ab: A | B) = {
          ab match {
            case A() => "Got A"
            case B() => "Got B"
          }
        }
        
        def test = {
          println("== Union ==")
          println(union(A()))
          println(union(B()))
          
          // println(union(C()))
          // Does not compile:
          // found:    Union.C
          // required: Union.A | Union.B
        }
      }

      object Main {
        def main(args: Array[String]): Unit = {
          Union.test
        }
      }
